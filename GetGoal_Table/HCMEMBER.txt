CREATE TABLE  "HCMEMBER" 
   (	"SEQ" NUMBER NOT NULL ENABLE, 
	"ID" VARCHAR2(20) NOT NULL ENABLE, 
	"NAME" VARCHAR2(40) NOT NULL ENABLE, 
	"PASSWORD" VARCHAR2(40) NOT NULL ENABLE, 
	"ADDRESS" VARCHAR2(2000) NOT NULL ENABLE, 
	"PHONE" VARCHAR2(20) NOT NULL ENABLE, 
	"EMAIL" VARCHAR2(40) NOT NULL ENABLE, 
	"ENABLED" VARCHAR2(2) NOT NULL ENABLE, 
	"ROLE" VARCHAR2(10) NOT NULL ENABLE, 
	"TTPOINT" NUMBER NOT NULL ENABLE, 
	"TDPOINT" VARCHAR2(40) NOT NULL ENABLE, 
	 CONSTRAINT "HCMEMBER_PK" PRIMARY KEY ("SEQ") ENABLE, 
	 CONSTRAINT "HCMEMBER_UK1" UNIQUE ("ID") ENABLE
   ) ;

CREATE OR REPLACE TRIGGER  "BI_HCMEMBER" 
  before insert on "HCMEMBER"              
  for each row 
begin  
  if :NEW."SEQ" is null then
    select "HCMEMBER_SEQ".nextval into :NEW."SEQ" from dual;
  end if;
end;
/
ALTER TRIGGER  "BI_HCMEMBER" ENABLE;
